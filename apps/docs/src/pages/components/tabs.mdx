## Usage

```tsx title="Imports"
import { Tabs, TabsContent, TabsList, TabsTrigger } from "@nui/core/tabs";
```

```tsx title="Anatomy"
<Tabs>
  <TabsList>
    <TabsTrigger />
  </TabsList>
  <TabsContent />
</Tabs>
```
