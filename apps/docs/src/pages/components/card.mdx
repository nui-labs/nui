## Usage

```tsx title="Imports"
import {
  Card,
  CardAction,
  CardContent,
  CardDescription,
  CardFooter,
  CardHeader,
  CardTitle,
} from "@nui/core/card";
```

```tsx title="Anatomy"
<Card>
	<CardHeader>
		<CardTitle />
		<CardDescription />
		<CardAction />
	</CardHeader>
	<CardContent>
	<CardFooter />
</Card>
```
