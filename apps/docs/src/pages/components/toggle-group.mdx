## Usage

```tsx title="Imports"
import { Toggle } from "@nui/core/toggle";
import { ToggleGroup } from "@nui/core/toggle-group";
```

```tsx title="Anatomy"
<ToggleGroup>
  <Toggle />
</ToggleGroup>
```
