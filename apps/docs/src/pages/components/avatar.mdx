## Usage

```tsx title="Imports"
import { Avatar, AvatarFallback, AvatarImage } from "@nui/core/avatar";
```

```tsx title="Anatomy"
<Avatar>
  <AvatarImage />
  <AvatarFallback />
</Avatar>
```
