## Usage

```tsx title="Imports"
import {
  Collapsible,
  CollapsibleContent,
  CollapsibleTrigger,
} from "@nui/core/collapsible";
```

```tsx title="Anatomy"
<Collapsible>
  <CollapsibleTrigger />
  <CollapsibleContent />
</Collapsible>
```
